# Code
```java []
        import java.util.*;
class Solution {
    public int[] constructRectangle(int area) {
        int a = (int)Math.sqrt(area);
        while(area%a !=0) a--;
        return new int[] {area/a,a};
    }
}
```
```python []
from math import sqrt
class Solution(object):
    def constructRectangle(self, area):
        """
        :type area: int
        :rtype: List[int]
        """
        s = int(sqrt(area))
        while(s > 1):
            if(area%s==0): 
                n=area//s
                arr = [s,n]
                arr.sort(reverse=True)
                return arr
            s-=1
        arr = [area,1]
        arr.sort(reverse=True)
        return arr
```        
